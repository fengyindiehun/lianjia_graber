function selectDateLast01(json,date){
var retr = -1;
for(var i= 0;i<json.length;i++){
if(json[i].ZNTS_OUT_DATE==date){
if(parseInt(json[i].ZNTS_CHENG_QTY)>parseInt(json[i].ZNTS_CHENG_SALE)){
return parseInt(json[i].ZNTS_CHENG_QTY)-parseInt(json[i].ZNTS_CHENG_SALE);
}
return 0;
}
}
return retr;
}
function selectDateLast02(json,date){

}

}
function getDateArrBymitinue(){

url: "/weixin/getChaoyangYuYueByWeixin.action",

data: {"sizeId":2020},
success: function(json){

//补全第一行不齐的
var currWeek = new Date().getDay();
var currDate = new Date().getDate();
var currYear = new Date().getFullYear();
var currMonth = new Date().getMonth()+1;
var lastMaxDay = new Date(new Date().getFullYear(),new Date().getMonth(),0).getDate();
var currMaxDay = new Date(new Date().getFullYear(),new Date().getMonth()+1,0).getDate();
//其实号数
var startDate = currDate - currWeek > 0 ? currDate - currWeek:lastMaxDay+currDate - currWeek;
var showMobth = "";
var showDate = "";
var showYear = currYear;
var systemSelect = 0;
for(var i = 0;i<4;i++){
for(var k = 0;k<7;k++){
if(currDate - currWeek > 0){
if(startDate+k+i*7<=currMaxDay){
showMobth = currMonth;
showDate = startDate+k+i*7;
}
else{
if(currMonth==12){
showMobth = 1;
showYear = currYear + 1;
}else{
showMobth = currMonth+1;
}
showDate = startDate+k+i*7-currMaxDay;
}
}
else{
if(startDate+k+i*7<=lastMaxDay){
showMobth = currMonth - 1;
showDate = startDate+k+i*7;
}else{
showMobth = currMonth;
showDate = startDate+k+i*7-lastMaxDay;
}
}
var checkedate = showYear+"-"+(showMobth<10?"0"+showMobth:showMobth)+"-"+(showDate<10?"0"+showDate:showDate);
var lastQty = selectDateLast01(json,checkedate);
if(lastQty==-1){
stl += "<div><span>"+showMobth+"-"+showDate+"</span><br/><span>不可约</span></div>";
}
else if(lastQty==0){
stl += "<div><span>"+showMobth+"-"+showDate+"</span><br/><span style='color:red;'>没号了</span></div>";
}
else{
var classText = "";
if(window.systemApply == 1&&$("div.active").length==0&&systemSelect == 0){
//系统自动刷单
if($("#system_select_date").val()==""||$("#system_select_date").val()==checkedate){
if($("#system_dateType").val()==""||$("#system_dateType").val()=="上午"){
classText = "class='active'";
systemSelect = 1;
selectOutDateBySystem(checkedate,"上午");
}
}
}
stl += "<div "+classText+" onclick='selectOutDate(this,\""+checkedate+"\",\"上午\")'><span>"+showMobth+"-"+showDate+"</span><br/><span style='color:#2a7e4e;'>余:"+lastQty+"</span></div>";
}
}
}
$("#dayArr_div_01").html(stl);
stl = "";
showYear = currYear;
for(var i = 0;i<4;i++){
for(var k = 0;k<7;k++){
if(currDate - currWeek > 0){
if(startDate+k+i*7<=currMaxDay){
showMobth = currMonth;
showDate = startDate+k+i*7;
}
else{
if(currMonth==12){
showMobth = 1;
showYear = currYear + 1;
}else{
showMobth = currMonth+1;
}
showDate = startDate+k+i*7-currMaxDay;
}
}
else{
if(startDate+k+i*7<=lastMaxDay){
showMobth = currMonth - 1;
showDate = startDate+k+i*7;
}else{
showMobth = currMonth;
showDate = startDate+k+i*7-lastMaxDay;
}
}
var checkedate = showYear+"-"+(showMobth<10?"0"+showMobth:showMobth)+"-"+(showDate<10?"0"+showDate:showDate);
var lastQty = selectDateLast02(json,checkedate);
if(lastQty==-1){
stl += "<div><span>"+showMobth+"-"+showDate+"</span><br/><span>不可约</span></div>";
}
else if(lastQty==0){
stl += "<div><span>"+showMobth+"-"+showDate+"</span><br/><span style='color:red;'>没号了</span></div>";
}
else{
var classText = "";
if(window.systemApply == 1&&$("div.active").length==0&&systemSelect == 0){
//系统自动刷单
if($("#system_select_date").val()==""||$("#system_select_date").val()==checkedate){
if($("#system_dateType").val()==""||$("#system_dateType").val()=="下午"){
classText = "class='active'";
systemSelect = 1;
selectOutDateBySystem(checkedate,"下午");
}
}
}
stl += "<div "+classText+" onclick='selectOutDate(this,\""+checkedate+"\",\"下午\")'><span>"+showMobth+"-"+showDate+"</span><br/><span style='color:#2a7e4e;'>余:"+lastQty+"</span></div>";
}
}
}
$("#dayArr_div_02").html(stl);
window.timeIndex ++;
$("#showTimeIndex").html("预约日期已为您刷新"+window.timeIndex+"次，若有余号，请及时提交订单！");
}
});
